public class TriggerHandlerEditSolRes {    
    List<Case_Solution__c> casetListUpd = new List<Case_Solution__c>();
    
    public void afterUpdateMethod(List<Case_Solution__c> caseSolNew){
        Case_Solution__c caseSolEdit = [SELECT ID, Solution_Results__c, Case__c 
                                        FROM Case_Solution__c 
                                        WHERE ID =: caseSolNew[0].Id LIMIT 1];
        
        if(caseSolEdit.Solution_Results__c == 'Resolved in advance' || caseSolEdit.Solution_Results__c == 'Resolved via call'){
            Case casRelat = [SELECT ID, Status, 
                             (SELECT ID, Solution_Results__c FROM Cases_Solution1__r) 
                             FROM Case 
                             WHERE ID =: caseSolEdit.Case__c];
            List<Case_Solution__c> caseColutList = casRelat.Cases_Solution1__r;
            
            casRelat.Status = 'Closed';
            
            for(Case_Solution__c cs : caseColutList){
                if(cs.Id != caseSolEdit.Id){
                    cs.Solution_Results__c = 'Cancelled';
                    casetListUpd.add(cs);
                }
            }
            
            update casRelat;
            update casetListUpd;
        }
        
    }
}
